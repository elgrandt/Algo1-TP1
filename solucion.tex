\begin{proc}{silencios}{in s: $\senial$, in prof : $\ent$, in freq: $\ent$, in umbral: $\amplitud$, out tiempos: \TLista{\intervalo}}{}
	\pre{grabacionValidaV(s,prof,freq) \land umbral > 0}
	
	\post{(\forall i: $\ent$) (0 \leq i < (j + (0.1 * freq)) < |s|) \land (todosBajosEntre(i,j,s,umbral) \Leftrightarrow estaDentro(tiempos,i,j))
	}
	\pred {todosBajosEntre} {s: $\senial$, a: $\ent$, b : $\ent$, umbral : $\amplitud$}{
	\newline
	(\forall i : $\ent$) \implica_L (|s[i]| < umbral) 
	}
	\pred {estaDentro}{tiempo : $\tiempo$, a : $\ent$, b : $\ent$}{
	\newline
	(\exists i : $\ent$) (0 \leq i < tiempo) \land_L ((a \geq tiempo[i]_0) \land (tiempo[i]_1 \geq b))	
	}	
	
\end{proc}


\begin{proc}{hayQuilombo}{in m : \TLista{$\senial$}, in prof:$\ent$, in freq : $\ent$, in umbral : $\amplitud$, out result : Bool}

	\pre{grabacionValidaM(m,prof,freq) \land umbral > 0 \land |m| > 0}
	\post{(\exists i : $\ent$)((0 \leq i < |m[0]|) \land_L cantHablantes(prof,freq,umbral,i) > 1 )}
	
	\aux{cantHablantes}{m :  \TLista{$\senial$}, prof : $\ent$, freq : $\ent$, umbral : $\amplitud$, instante : $\ent$}{res : $\ent$}{
	\sum_{i=0}^{|m|-1} estaHablando(m[i], instante, freq,umbral)
	}
	\aux{estaHablando}{s: $\senial$, freq:$\ent$, umbral : $\amplitud$}{res : $\ent$} {
	\newline
		ifThenElseFi( (\exists i : $\ent$)(\exists j : $\ent$) (0 \leq i \leq instante \leq j < |s|) \land (j-i \geq 0.1*freq) \land todosBajosEntre(s,u,j,umbral),0,1)
	}
	
\end{proc}
