\begin{proc}{reacomodar}{inout m: \TLista{$\senial$}, in prof: $\ent$, in freq $\ent$}{}
	\pre{(m = m_0 \land esMatriz(m) \land esGrabacionValida(m, freq, prof)) } 
	
	\post{( (\forall i:$\ent$) (0 \leq i < |m_0| - 1 \implica_L (calcIntensidadMedia(m[i]) \leq calcIntensidadMedia(m[i+1])) ) ) \land \newline (esMatriz(m) \land esGrabacionValida(m, freq, prof) \land igualApariciones(m, m_0))}
	\aux{calcIntensidadMedia}{in s: $\senial$, in prof : \ent}{\ent}{\sum_{i=0}^{|s|-1}{\frac{vAbs(s[i])}{|s|-1}}}
	\aux{vAbs}{in valor:\ent}{\ent} {IfThenElseFi(valor <0,valor,-valor)}
	\pred{esGrabacionValida}{m: \TLista{$\senial$}, freq : $\ent$, prof : $\ent$}{
	\newline
	enRango(m, prof) \land duracionMayorA10(m, freq) \land micFunciona(m, freq) \land profValida(prof) \land freqValida(freq)
	}
	\pred {esMatriz}{m:\TLista{$\senial$}, prof:$\ent$}{
	\newline		
		((\forall i:$\ent$)(\forall j:$\ent$)(0 \leq i < |m| \implica_L |m[i]| = m[j]))
	}
	\pred {enRango}{m: \TLista{$\senial$}, prof:$\ent$}{
	\newline	
		(\forall i:$\ent$)(\forall j:$\ent$)(0 \leq i < |m| \land_L 0 \leq j < |m[i]|) \implica_L (-2^{(prof-1)} \leq m[i][j] < (2^{(prof-1)})-1)
	}

	\pred {duracionMayorA10}{m:\TLista{$\senial$}, freq:$\ent$}{
	\newline
		(\forall i:$\ent$)(0 \leq i < |m[i]| \implica_L 10*freq > |m[i]|)
	}
	\pred {micFunciona}{m:\TLista{$\senial$}, freq:$\ent$}{
	\newline
	(\forall i:$\ent$)(\nexists j:$\ent$)(0 \leq i < |m| \implica_L $\#$apariciones(subsec(m[i]),j,j+freq),0) = freq
	}
	\pred{profValida}{prof: $\ent$}{
		(prof = 16)\lor(prof = 32)
	}
	\pred{freqValida}{freq: $\ent$}{
		(freq = 44100)\lor(freq = 48000)
	}
	\pred{igualApariciones}{m:\TLista{$\senial$}, n:\TLista{$\senial$}}{
	\newline
		(\forall i : $\ent$)(\exists j: $\ent$)(0 \leq i < |n| \implica_L n[i]=m[j])
		}	
\end{proc}

\begin{proc}{calmateJose!}{in s:$\senial$, in umbralHulk:amplitud, in prof:$\ent$, in freq : $\ent$, out result Bool} {}
	\pre{|s| > 0 \land grabacionCorrecta(s, prof, freq)}
	\post{((\exists i: $\ent$)(\exists j: $\ent$)(\exists k : $\ent$)((k-i) > freq) \land (({\sum_{q=0}^{i}{\frac{s[q]}{i}}} < umbralHulk ) \land ({\sum_{q=i}^{k}{\frac{s[q]}{k-i}}} > umbralHulk) \land ({\sum_{q=j}^{|s|}{\frac{s[q]}{|s|-j}}}) < umbralHulk)) )\Leftrightarrow result = \True
	}
	\pred{grabacionValida}{s:$\senial$, prof:$\ent$, freq : $\ent$}{
	\newline
		profValida(prof) \land freqValida(freq) \land enRango(s, prof) \land duracionValida(s,freq) \land micFunciona(s,freq)	
	}
	\pred {enRango}{s:$\senial$, prof:$\ent$}{
	\newline
	(\forall i:$\ent$)(0 \leq i < |s| \implica_L (-2^{(prof-1)} \leq s[i] < 2^{(prof-1)}-1) )	
	}
	\pred{duracionValida}{s:$\senial$, freq:$\ent$}{
	\newline
	|s| > 10 * freq 	
	}
	\pred{micFunciona}{s:$\senial$, freq:$\ent$}{
	\newline
	(\nexists j : $\ent$)($\#$apariciones(subsec(s,j,j+freq+1),0) = freq)
	}
	\pred{profValida}{prof: $\ent$}{
		(prof = 16)\lor(prof = 32)
	}
	\pred{freqValida}{freq: $\ent$}{
		(freq = 44100)\lor(freq = 48000)
	}
\end{proc}

\begin{proc}{ardillizar}{inout m: \TLista{$\senial$}, in prof: $\ent$, in freq: $\ent$}{}
	\pre{se\tilde{n}alEsValida(m, prof, freq) \land m = m_0}
	\post{se\tilde{n}alEsValida(m, prof, freq) \land contieneMuestrasCorrectas(m, m_0)\land largoDeSe\tilde{n}alCorrecto(m, m_0)}
	\pred{contieneMuestrasCorrectas}{m: \TLista{$\senial$}, n: \TLista{$\senial$}}{
		\newline
		(\forall i: $\ent$)(0 \leq i < |m|)\implicaLuego(validarMuestras(m[i], n[i]))\newline
	}
	\pred{validarMuestras}{s: $\senial$, so: $\senial$}{
		(\forall i: $\ent$)(0 \leq i < \frac{|so|}{2})\implicaLuego(s[i]=so[i*2])
	}
	\pred{largoDeSe\~nalCorrecto}{m: \TLista{$\senial$}, n: \TLista{$\senial$}}{
		(\forall i: $\ent$)(0 \leq i < |m|)\implicaLuego(|m[i]| = \frac{|n[i]|}{2})
	}
\end{proc}

\begin{proc}{flashElPerezoso}{inout m: \TLista{$\senial$}, in prof: $\ent$, in freq: $\ent$}{}
	\pre{se\tilde{n}alEsValida(m, prof, freq) \land m = m_0}
	\post{tieneLongitudCorrecta(m, m_0) \land tieneMuestrasCorrectas(m, m_0)}
	\pred{tieneLongitudCorrecta}{m: \TLista{$\senial$}, n: \TLista{$\senial$}}{
		\newline
		(\forall i: $\ent$)(0 \leq i < |m|)\implicaLuego(|m[i]|=|n[i]|+(|n[i]|-1))
	}
	\pred{tieneMuestrasCorrectas}{m: \TLista{$\senial$}, n: \TLista{$\senial$}}{
		\newline
		(\forall i: $\ent$)(0 \leq i < |m|)\implicaLuego(validarMuestras(m[i], n[i]))
	}
	\pred{validarMuestras}{s: $\senial$, or: $\senial$}{
		\newline
		((\forall i: $\ent$)(0 \leq i < |or|)\implicaLuego(or[i] = s[i*2])) \newline
		\land ((\forall i: $\ent$)(0 \leq i < |or|-1)\implicaLuego(s[i*2+1] = \frac{or[i]+or[i+1]}{2}))
	}
\end{proc}

\begin{proc}{silencios}{in s: $\senial$, in prof : $\ent$, in freq: $\ent$, in umbral: $\amplitud$, out tiempos: \TLista{\intervalo}}{}
	\pre{grabacionValidaV(s,prof,freq) \land umbral > 0}
	
	\post{(\forall i: $\ent$) (0 \leq i < (j + (0.1 * freq)) < |s|) \land (todosBajosEntre(i,j,s,umbral) \Leftrightarrow estaDentro(tiempos,i,j))
	}
	\pred {todosBajosEntre} {s: $\senial$, a: $\ent$, b : $\ent$, umbral : $\amplitud$}{
	\newline
	(\forall i : $\ent$) \implica_L (|s[i]| < umbral) 
	}
	\pred {estaDentro}{tiempo : $\tiempo$, a : $\ent$, b : $\ent$}{
	\newline
	(\exists i : $\ent$) (0 \leq i < tiempo) \land_L ((a \geq tiempo[i]_0) \land (tiempo[i]_1 \geq b))	
	}	
	
\end{proc}

	\pre{grabacionValidaM(m,prof,freq) \land umbral > 0 \land |m| > 0}

	\sum_{i=0}^{|m|-1} estaHablando(m[i], instante, freq,umbral)
	}

	}
	
\end{proc}

\pred{se\~nalEsValida}{m: \TLista{$\senial$}, prof: $\ent$, freq: $\ent$}{
	(\forall s: $\senial$)(s \in m)$\implica$(validarSe\tilde{n}al(s, prof, freq))
}

\pred{validarSe\~nal}{s: $\senial$, prof: $\ent$, freq: $\ent$}{
	\newline
	profundidadCorrecta(prof) \newline
	\land frecuenciaCorrecta(freq) \newline
	\land duracionDeSe\tilde{n}alCorrecta(s, freq) \newline
	\land amplitudesDentroDelRango(s, prof) \newline
	\land microfonoFunciona(s, freq) \newline
}

\pred{profundidadCorrecta}{prof: $\ent$}{
	(prof = 16)\lor(prof = 32)
}
	
\pred{frecuenciaCorrecta}{freq: $\ent$}{
	(freq = 44100)\lor(freq = 48000)
}

\pred{duracionDeSe\~nalCorrecta}{s: $\senial$, freq: $\ent$}{
	(|s| > 10 * freq)
}

\pred{amplitudesDentroDelRango}{s: $\senial$, prof: $\ent$}{
	\newline
	(\forall amp: $\ent$)(amp \in s)$\implica$(-2^{(prof-1)} \leq amp < (2^{(prof-1)})-1 ) \newline
}

\pred{microfonoFunciona}{s: $\senial$, freq: $\ent$}{
	\newline
	\lnot((\exists i: $\ent$)(\exists j: $\ent$)((0 \leq i < j \leq |s|) \land_L ((|subseq(s, i, j)| \geq freq)\newline
	\land (\forall k: $\ent$)(k \in subseq(s, i, j)$\implica$(k=0))))\newline
}
